server {
    listen 80;
    server_name localhost;

    root /usr/share/nginx/html;
    index index.html;

    # Adiciona a configuração de tipos MIME.
    # Isso ensina o Nginx a enviar o Content-Type correto para .js, .css, etc.
    # E corrige o erro "application/octet-stream".
    include /etc/nginx/mime.types;

    location / {
        # Tenta servir o arquivo solicitado diretamente.
        # Se não encontrar, redireciona a requisição para /index.html (para o React Router).
        try_files $uri $uri/ /index.html;
    }
}
